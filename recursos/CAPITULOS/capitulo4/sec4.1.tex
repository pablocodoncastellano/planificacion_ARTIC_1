
\section{Líderes de Proyecto (Ingeniería de Sistemas)}
\begin{itemize}
    \item \textbf{Responsabilidad:} Definir los Requisitos del Sistema (SRD) y el Power Budget inicial.
    \item \textbf{Enfoque:} Tomar decisiones a nivel de sistema, asegurar interfaces y validar el cumplimiento del ICD.
\end{itemize}

\section{Equipo de la Asignatura (Contratista de Hardware)}
\begin{itemize}
    \item \textbf{Responsabilidad:} Ejecutar el diseño físico y matemático. Su motivación está vinculada a la evaluación académica.
    \item \textbf{Enfoque:} Selección de componentes basada en el Power Budget, simulaciones en PSIM, esquemáticos en KiCad y memorias justificativas.
\end{itemize}

\section{Equipo SART (Arquitectos de Lógica y Misión)}
\begin{itemize}
    \item \textbf{Responsabilidad:} Diseño del comportamiento inteligente del satélite y comandos.
    \item \textbf{Enfoque:} Máquina de estados, interfaces lógicas con OBC, frecuencias y modos de vuelo.
\end{itemize}

\section{Agentes Dobles (Especialistas de Carga Útil/Estructuras)}
\begin{itemize}
    \item \textbf{Responsabilidad:} Actualizar sus respectivos módulos del proyecto anterior para que encajen con los nuevos Requisitos.
\end{itemize}

\chapter{Línea de Tiempo de Tareas y Dependencias}

\noindent Esta es la ruta crítica técnica basada en los Paquetes de Trabajo (WP). El avance está condicionado a la aprobación del entregable anterior:

\begin{enumerate}
    \item \textbf{Definición de Misión [WP 1000] y Especificaciones Mecánicas [WP 3020]:} Establecer la base del diseño y el sensor de carga útil.
    \item \textbf{Especificaciones Eléctricas [WP 3030]:} Cierre del Excel de consumos (\textit{Power Budget}).
    \item \textbf{Búsqueda de Componentes [WP 2030]:} Selección comercial basada en el presupuesto de potencia.
    \item \textbf{Ensayos Preliminares de Validación [WP 2040]:} Simulación en PSIM (Fase teórica y Fase realista con protecciones).
    \item \textbf{Especificaciones Operacionales y Generales [WP 3040 / 3010]:} Diagramas de estados y definición de interfaces (ICD).
    \item \textbf{Diseño de Unidad de Vuelo [WP 2020]:} Captura formal de esquemáticos en KiCad.
    \item \textbf{Requisitos de Ensayo [WP 4000]:} Redacción en paralelo del documento teórico de validación ambiental.
\end{enumerate}

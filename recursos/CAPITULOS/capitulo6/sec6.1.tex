\section{Mes 1: Requisitos, Presupuesto y Selección}
\begin{itemize}
    \item \textbf{Semana 1:} Líderes fijan el \textbf{SRD}. SART y Líderes elaboran y cierran el \textbf{Power Budget Master} (Entregable crítico).
    \item \textbf{Semana 2:} Los Agentes Dobles entregan el \textbf{Mechanical ICD} (restricciones de tamaño). Asignatura realiza el \textbf{Trade-Off Report} basándose estrictamente en los datos del Power Budget.
    \item \textbf{Semana 3:} Asignatura presenta la lista de la compra. Líderes aprueban el \textbf{BOM Preliminar}.
    \item \textbf{Semana 4:} Asignatura integra los modelos matemáticos en PSIM y entrega el \textbf{PSIM Simulation Report v1.0}. SART comienza a esbozar la lógica de vuelo.
\end{itemize}

\section{Mes 2: Simulación Avanzada y Definición de Interfaces}
\begin{itemize}
    \item \textbf{Semana 5:} Asignatura añade componentes reales, Load Switches y picos de arranque, entregando el \textbf{PSIM Simulation Report v2.0}. 
    \item \textbf{Semana 6:} SART finaliza y entrega el \textbf{State Machine Diagram}. Revisión de Integración conjunta: ¿El hardware simulado soporta los estados lógicos?
    \item \textbf{Semana 7:} SART recopila pines, comandos y frecuencias para publicar el \textbf{Interface Control Document (ICD)} definitivo.
    \item \textbf{Semana 8:} Con el ICD y el BOM cerrados, Asignatura inicia el diseño en ECAD (creación de símbolos y huellas personalizadas).
\end{itemize}

\section{Mes 3: Esquemáticos y Cierre}
\begin{itemize}
    \item \textbf{Semana 9:} Asignatura rutea conexiones en KiCad estructurando por bloques jerárquicos y genera el primer borrador de los planos.
    \item \textbf{Semana 10: Critical Design Review (CDR).} Reunión conjunta de auditoría. SART y Líderes revisan exhaustivamente el KiCad frente al SRD y los Riesgos. Se redacta y firma el \textbf{CDR Acta}.
    \item \textbf{Semana 11:} Asignatura corrige los defectos identificados en la CDR (ej. pines cruzados, huellas incorrectas) y aprueba el ERC (Electrical Rules Check).
    \item \textbf{Semana 12:} Consolidación final. Asignatura entrega el \textbf{Schematics Package} completo. Preparación de presentaciones académicas y cierre del hito.
\end{itemize}

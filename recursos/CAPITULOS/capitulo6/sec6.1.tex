
\section{Mes 1: Requisitos, Presupuesto y Selección}
\begin{itemize}
    \item \textbf{Semana 1:} \textbf{[WP 1010]} SART realiza la búsqueda del sensor de basura espacial comercial para extraer sus requerimientos de consumo. Líderes fijan el \textbf{SRD [WP 1000]}.
    \item \textbf{Semana 2:} SART y Líderes cierran el \textbf{Power Budget [WP 3030]}. Agentes Dobles entregan el \textbf{Mechanical ICD [WP 3020]}.
    \item \textbf{Semana 3:} Asignatura realiza la búsqueda de materiales y entrega el \textbf{Trade-Off Report y BOM [WP 2030]}. 
    \item \textbf{Semana 4:} Asignatura integra los modelos matemáticos en PSIM y entrega la primera fase de los \textbf{Ensayos Preliminares [WP 2040]}.
\end{itemize}

\section{Mes 2: Simulación Avanzada y Definición de Interfaces}
\begin{itemize}
    \item \textbf{Semana 5:} Asignatura añade componentes reales y picos de arranque a PSIM, cerrando la validación del \textbf{[WP 2040]}. SART esboza los estados lógicos.
    \item \textbf{Semana 6:} SART finaliza las Especificaciones Operacionales entregando el \textbf{State Machine Diagram [WP 3040]}. 
    \item \textbf{Semana 7:} SART recopila pines y comandos para publicar el \textbf{Interface Control Document [WP 3010]}. En paralelo, inician la redacción teórica del \textbf{Test Philosophy Report [WP 4000]}.
    \item \textbf{Semana 8:} Con el ICD y el BOM cerrados, Asignatura inicia el Diseño de la Unidad de Vuelo \textbf{[WP 2020]} en ECAD.
\end{itemize}

\section{Mes 3: Esquemáticos y Cierre}
\begin{itemize}
    \item \textbf{Semana 9:} Asignatura rutea conexiones eléctricas en KiCad estructurando por bloques.
    \item \textbf{Semana 10: Critical Design Review (CDR).} Auditoría conjunta del KiCad frente a los requisitos. SART entrega la versión final del documento de ensayos \textbf{[WP 4000]}.
    \item \textbf{Semana 11:} Asignatura corrige los defectos identificados en la CDR y aprueba el chequeo de reglas eléctricas (ERC).
    \item \textbf{Semana 12:} Consolidación final. Asignatura entrega el \textbf{Schematics Package [WP 2020]} completo. Preparación de memorias y presentación final antes del examen (16 de mayo).
\end{itemize}
